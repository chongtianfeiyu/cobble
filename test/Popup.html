Popup

实例参数

    {
        element: $('xx'),               // 弹出显示的元素
        trigger: $('yy'),               // 可选，触发显示的元素
                                        //     如果不需要交互触发，如 popup.show() 出来，可无视
        showBy: 'click',                // 可选，触发显示的方式，注意 trigger 必须有值
                                        //     可选值包括 'click', 'over'
        hideBy: 'blur',                 // 可选，触发隐藏的方式
                                        //     可选值包括 'blur', 'out', 'blur,out'
        showDelay: 100,                 // 可选，showBy 为 over 时的延时显示时间
        hideDelay: 200,                 // 可选，hideBy 包括 out 时的延时隐藏时间
        show: function () {},           // 可选，显示函数
                                        //     默认是 element.show()，如果有特效需求，自行实现
        hide: function () {},           // 可选，隐藏函数
                                        //     默认是 element.hide()，如果有特效需求，自行实现
        onBeforeShow: function () {},   // 可选，返回 false 可阻止显示
        onAfterShow: function () {},    // 可选
        onBeforeHide: function () {},   // 可选，返回 false 可阻止隐藏
        onAfterHide: function () {}     // 可选
    }

使用方法

    var popup = new Popup(options);
    popup.dispose();
