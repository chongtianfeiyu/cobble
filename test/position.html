定位

pin 参数

    {
        element: $('xx'),        // 需要定位的元素
        attachment: $('yy'),     // 可选，参照物元素
                                 //     默认是 $(document.body)

        x: 10,                   // 定位元素的定位点横坐标
                                 // 可选值包括 'left', 'center', 'right', 'xx%'(百分比), xx(数字)
        y: 10,                   // 定位元素的定位点纵坐标
                                 // 可选值包括 'top', 'middle', 'bottom', 'yy%'(百分比), yy(数字)

        attachmentX: 10,         // 参照物元素的参照点横坐标，取值同 x
        attachmentY: 10,         // 参照物元素的参照点纵坐标，取值同 y

        offsetX: 0,              // 可选，定位元素的横坐标偏移量
        offsetY: 0               // 可选，定位元素的纵坐标偏移量
    }

使用方法

    // 真正的定位函数，可使用上面所有配置
    position.pin(options);

    // 下面是一些快捷方法
    // 只需要设置 element 和 attachment 即可 ( 或配合 offsetX/offsetY 使用 )
    position.topLeft(options);
    position.topCenter(options);
    position.topRight(options);

    position.middleLeft(options);
    position.middleCenter(options);
    position.middleRight(options);

    position.bottomLeft(options);
    position.bottomCenter(options);
    position.bottomRight(options);
